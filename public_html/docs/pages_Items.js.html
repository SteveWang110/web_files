        fetch(backend_url + 'items')
          .then(response => {
            return response.text();
          })
          .then(data => {
            setItems(data);
          });

          method: 'POST',
          headers: {  
            'Content-Type': 'application/json',
          },
          body: JSON.stringify({price_amount, name, id}),
        })
          .then(response => {
            return response.text();
