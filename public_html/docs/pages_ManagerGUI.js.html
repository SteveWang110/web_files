/**
 * Displays Manager GUI page
 */
 function ManagerGUI() {
    let component
    switch (window.location.pathname) {
        case "/about":
          component = <About/>
          break;
        case "/tranHistory":
          component = <Transaction/>
          break;
        case "/employees":
          component = <Employees/>
          break;
        case "/items":
          component = <Items/>
          break;
        case "/reports":
          component = <Reports/>
          break;
        case "/Inventory":
          component = <Inventory/>
          break;

        default: 
          component = <Login/>
          return <Login/>;
          
      }
      return (
        <>
          <NavBarManager/>
          {component}
        </>
      )
    

}   
export default ManagerGUI;
