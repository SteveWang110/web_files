      const { employeeid, salary, name, managerid, role } = body
      pool.query('INSERT INTO employees (employeeid, salary, name, managerid, role) VALUES($1, $2, $3, $4, $5) RETURNING *', [employeeid, salary, name, managerid, role], (error, results) => {

          reject(error)

      })
    })
